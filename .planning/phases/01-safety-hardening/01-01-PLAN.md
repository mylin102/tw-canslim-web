    stock_id: str,
    stock_entry: dict,

    required_paths: tuple[str, ...] = REQUIRED_STOCK_PATHS,

def publish_artifact_bundle(
    bundle: dict[str, dict],
    *,
    lock_path: str = "docs/.publish.lock",

    logger=None,
    json_default=None,

def restore_latest_bundle(
    *,
    lock_path: str = "docs/.publish.lock",
    backup_dir: str = "backups/last_good",
    logger=None,
    targets: tuple[str, ...] | None = None,

## Trust Boundaries

| Boundary | Description |
|----------|-------------|

| concurrent writer→artifact bundle | Multiple local or GitHub Actions processes compete to publish related `docs/` artifacts |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | T | `publish_safety.py` validation | mitigate | Validate each payload by `artifact_kind` before any staged file is promoted. |
| T-01-02 | D | bundle publish lock | mitigate | Use one repo-level publish lock for the related artifact bundle instead of per-target locks. |
| T-01-03 | R | resume compatibility | mitigate | Reject incompatible records with explicit validation errors so skipped stocks are auditable. |
| T-01-04 | R | backup/restore manifest | mitigate | Persist manifest metadata for each validated bundle and restore from the latest manifest-backed snapshot only. |
